stateAfterPatientRegistration strAge strWeight = guiNext patientRegistrationState (textboxInput2 strAge  strWeight)

lemStateAfterPatientRegistration : (strAge strWeight : String) → stateAfterPatientRegistration strAge strWeight ≡
                                                    patientHistoryState (string2AgeCat strAge)
