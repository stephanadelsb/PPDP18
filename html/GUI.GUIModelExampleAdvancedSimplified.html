BtnFrameStart : Frame
BtnFrameStart = addButton "Start"  emptyFrame

BtnFrameXRay : Frame
BtnFrameXRay = addButton "X-Ray" emptyFrame

BtnFrameTreatment : Frame
BtnFrameTreatment = addButton "Treatment" emptyFrame


mutual
 startGUI : ∀ i → GUI {i}
 startGUI i .gui = BtnFrameStart
 startGUI i .obj .method {j} m = returnGUI (GUIXRay j)

 GUIXRay : ∀ i → GUI {i}
 GUIXRay i .gui = BtnFrameXRay
 GUIXRay i .obj .method {j} m =
   returnGUI (GUITreatment j)

 GUITreatment : ∀ i → GUI {i}
 GUITreatment i .gui = BtnFrameTreatment
 GUITreatment i .obj .method {j} m =
   returnGUI (startGUI i)

-- states
--
stateStart : GuiState
stateStart = state (startGUI ∞) notStarted

stateXRay : GuiState
stateXRay = state (GUIXRay ∞) notStarted

stateTreatment : GuiState
stateTreatment = state  (GUITreatment ∞) notStarted



-- next state start
--
corstateStartNext1∀  : (m : GuiState)
                       → (stateStart -gui->¹ m)
                       →  m ≡ stateXRay
corstateStartNext1∀ m (step _) = refl


corstateStartNext∃   : stateStart -gui->¹ stateXRay
corstateStartNext∃ =  step (zero , _)


-- next state xray
--
corstateXRayNext1∀  : (m : GuiState)
                       → (stateXRay -gui->¹ m)
                       →  m ≡ stateTreatment
corstateXRayNext1∀ m (step _) = refl

corstateTreatmentNext∃   : stateTreatment -gui->¹ stateStart
corstateTreatmentNext∃ =  step (zero , _)

goesThruXray :  (path : stateStart -gui-> stateTreatment)
                →  path goesThru stateXRay
goesThruXray (step c p) = inj₂ (goesThruSelf p)


stateTreatmentReachesEventuallyTreatment : stateTreatment -eventually-> stateTreatment
stateTreatmentReachesEventuallyTreatment = hasReached

stateXrayReachesEventuallyTreatment : stateXRay -eventually-> stateTreatment
stateXrayReachesEventuallyTreatment = next λ _ →
                                      stateTreatmentReachesEventuallyTreatment


stateStartReachesEventuallyTreatment : stateStart -eventually-> stateTreatment
stateStartReachesEventuallyTreatment = next λ _ →
                                   stateXrayReachesEventuallyTreatment
