-- 1) draw and receive blood value
--    --> get "renalValue"
--

-- 2) give either:
-- medicationOld  [if renalVal <30]
-- medicationNew  [if renalVal >= 30]

-- 3) final state: discharge

oneButtonFrame : String → Frame
oneButtonFrame str = addButton str  emptyFrame

twoButtonFrame : String → String → Frame
twoButtonFrame str1 str2 = addButton str2 (oneButtonFrame str1)

--

frmDrawBlood : Frame
frmDrawBlood = addButton "draw" (addLabel "renalValue" emptyFrame)

frmDischarge : Frame
frmDischarge = (addLabel "Finished" emptyFrame)


RenalValue : Set
RenalValue = ℕ

data RenalValueEval : Set where
  low high : RenalValueEval

postulate string2RenalValue : String → RenalValue

postulate evalRenalValue : RenalValue → RenalValueEval


frmOldMedication : Frame
frmOldMedication = oneButtonFrame "OldMedication"

frmNewMedication : Frame
frmNewMedication = oneButtonFrame "NewMedication"


frmDecideMedicationaux : RenalValueEval → Frame
frmDecideMedicationaux low = frmOldMedication
frmDecideMedicationaux high = frmNewMedication

frmDecideMedication : RenalValue → Frame
frmDecideMedication n = frmDecideMedicationaux  (evalRenalValue n)

mutual
 drawBloodGUI : ∀ i → GUI {i}
 drawBloodGUI i .gui = frmDrawBlood
 drawBloodGUI i .obj .method {j} m =
   exec (putStrLn "What is the renal value?") λ _ →
   exec getLine λ s →
   returnGUI (decideMedicationGUI i (string2RenalValue s))

 oldMedicationGUI : ∀ i → GUI {i}
 oldMedicationGUI i .gui = frmOldMedication
 oldMedicationGUI i .obj .method {j} (zero , _) = returnGUI (dischargeGUI j)
 oldMedicationGUI i .obj .method {j} (suc () , g_)

 newMedicationGUI : ∀ i → GUI {i}
 newMedicationGUI i . gui = frmNewMedication
 newMedicationGUI i .obj .method {j} (zero  , _ ) = returnGUI (dischargeGUI j)
 newMedicationGUI i .obj .method {j} (suc () , _ )


 decideMedicationauxGUI : ∀ i → (rv : RenalValueEval) → GUI {i}
 decideMedicationauxGUI i rv .gui = frmDecideMedicationaux rv
 decideMedicationauxGUI i low .obj  = oldMedicationGUI i .obj
 decideMedicationauxGUI i high .obj = newMedicationGUI i .obj

 decideMedicationGUI : ∀ i → (rv : RenalValue) → GUI {i}
 decideMedicationGUI i rv .gui = frmDecideMedication rv
 decideMedicationGUI i rv .obj = decideMedicationauxGUI i (evalRenalValue rv) .obj


 dischargeGUI : ∀ i → GUI {i}
 dischargeGUI i .gui = frmDischarge
 dischargeGUI i .obj .method {j} ()




-- states


stateDrawBlood : GuiState
stateDrawBlood = state  (drawBloodGUI ∞) notStarted


-- state we reach after we pressed the button which is button number zero so
-- zero is supplied

-- the textbox is here only used for displaying information, it is not relevant.
stateDrawBlood₁ : GuiState
stateDrawBlood₁ = guiNext stateDrawBlood (zero , tt)


-- once we have pressed the button it prints something out on console. That
--    step is ignored
-- then we get a string, we need to provide it in order to get into the next step

stateDrawBlood₂ : String → GuiState
stateDrawBlood₂ s = guiNext stateDrawBlood₁ s

stateOldMedication : GuiState
stateOldMedication = state (oldMedicationGUI ∞ ) notStarted

stateNewMedication : GuiState
stateNewMedication = state (newMedicationGUI ∞ ) notStarted

stateDecideMedicationEval : RenalValueEval → GuiState
stateDecideMedicationEval e =  state  (decideMedicationauxGUI ∞ e) notStarted


stateDecideMedication : RenalValue → GuiState
stateDecideMedication r =  state  (decideMedicationGUI ∞ r) notStarted


stateDischarge : GuiState
stateDischarge = state  (dischargeGUI ∞) notStarted


-- the state after input of a string is
-- obtained by first pressing the button (number zero) therefore first input zero
-- then by providing a string as input to the get String command) argument s
-- and get into state   (stateAfterInput s)
stateAfterInput : String → GuiState
stateAfterInput s = guiNext (guiNext stateDrawBlood (zero , tt)) s
